---
title: "Store feedback"
api: "POST https://api.usefini.com/api/v1/feedback"
description: "This endpoint allows you to store feedback from answers in the /answer endpoint"
---

### Headers

<ParamField body="token" type="string" required="true">
  This is the unique api-key for your bot 
</ParamField>

### Body

<ParamField body="answer_uuid" type="string" required="true">
 This is the anwer_uuid retrieved from /answer response object
</ParamField>

<ParamField body="feedback_type" type="string" required="true">
  This is the feedback type from users. It can be set to [“thumbs_up”, “thumbs_down”]
</ParamField>


### Response

<ResponseField name="message" type="string">
  Indicates whether the feedback was added.
</ResponseField>


<RequestExample>

```python Example Request- Python
import requests
endpoint = "https://api.usefini.com/api/v1/feedback"

answer_uuid = <uuid> # Replace this with the retrieved answer_uuid from /answer endpoint
feedback_type = "thumbs_up" # or "thumbs_down"
data = {“answer_uuid”: answer_uuid, “feedback_type”: feedback_type}

token = "your_api_key_here"  # Best stored in secrets management
headers = {"Authorization": "Bearer " + token}

# Make sure your ‘content-type’ header is set to ‘application/json’
response = requests.post(endpoint, json=data, headers=headers)
print(response.json());
```

```javascript Example Request- Node.js
const axios = require('axios');  // Make sure to install axios: npm install axios

const endpoint = 'https://api.usefini.com/api/v1/feedback';

const answer_uuid = <uuid>; // Replace this with the retrieved answer_uuid from /answer endpoint
const feedback_type = "thumbs_up" //or "thumbs_down";

const data = { answer_uuid: answer_uuid, feedback_type: feedback_type }; 

const token = 'your_api_key_here'; // Best stored in secrets management

const headers = {
    Authorization: 'Bearer ' + token,
    'Content-Type': 'application/json'
};

axios.post(endpoint, data, { headers: headers })
    .then((response) => {
        console.log(response.data);
    })
    .catch((error) => {
        console.error(error);
    });
```
