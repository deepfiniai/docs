---
title: "List Knowledge Items"
api: "GET https://api-prod.usefini.com/v2/knowledge-items/public"
description: "This endpoint fetches knowledge items from your Fini workspace based on filters such as status, keyword, category, or escalation settings."
---

### Query Parameters

<ParamField body="status" type="string">
  Filter knowledge items by status. Valid values: `LIVE`, `IN_REVIEW`, `DRAFT`.
</ParamField>

<ParamField body="question" type="string">
  Filter items containing this question
</ParamField>

<ParamField body="answer" type="string">
  Filter items containing this answer
</ParamField>

<ParamField body="keyword" type="string">
  Filter items that include the specified keyword.
</ParamField>

<ParamField body="category" type="string">
  Filter items under a specific category.
</ParamField>

<ParamField body="subcategory" type="string">
  Filter items under a specific subcategory.
</ParamField>

<ParamField body="escalation" type="boolean">
  Filter items marked for escalation
</ParamField>

<ParamField body="instruction" type="string">
  Filter items by the associated instruction content
</ParamField>

### Headers

<ParamField body="Authorization" type="string" required>
  API key for authentication. Format: `Bearer your_api_key_here`.
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates if the API call was successful.
</ResponseField>

<ResponseField name="data" type="array">
  List of matching knowledge items.
  
  <Expandable title="Toggle object">
    
    <ResponseField name="question" type="string">
      The question text of the knowledge item.
    </ResponseField>

    <ResponseField name="answer" type="string">
      The answer associated with the question.
    </ResponseField>

    <ResponseField name="status" type="string">
      Status of the item: `LIVE`, `IN_REVIEW`, or `DRAFT`.
    </ResponseField>

    <ResponseField name="category" type="string">
      Category under which the item is grouped.
    </ResponseField>

    <ResponseField name="subcategory" type="string">
      Subcategory under which the item is grouped.
    </ResponseField>

    <ResponseField name="escalation" type="boolean">
      Whether the item is marked for escalation.
    </ResponseField>

    <ResponseField name="instruction" type="string">
      Instruction associated with the item.
    </ResponseField>

  </Expandable>
</ResponseField>

<RequestExample>

```javascript Example Request- Node.js
const axios = require('axios');

const url = 'https://api-prod.usefini.com/v2/knowledge-items/public';
const token = 'your_api_key_here'; // Replace with your actual API key

const headers = {
    'authorization': `Bearer ${token}`
};

const params = {
    status: 'LIVE', // or IN_REVIEW, DRAFT
    question: 'upgrade',
    answer: 'premium plan',
    keyword: 'subscription',
    category: 'Pricing',
    subcategory: 'Plans',
    escalation: false,
    instruction: 'Answer questions related to pricing clearly.'
};

axios.get(url, { headers, params })
    .then(response => {
        console.log(response.data);
    })
    .catch(error => {
        console.error('Error:', error);
    });
```
</RequestExample>
