---
title: "Update Knowledge Item"
api: "PUT https://api-prod.usefini.com/v2/knowledge-items/:id/public"
description: "This endpoint updates an existing knowledge item in your Fini workspace by ID."
---

### Headers

<ParamField body="Authorization" type="string" required>
  API key for authentication. Format: Bearer your_api_key_here.
</ParamField>

### Path Parameters

<ParamField body="id" type="string" required>
  The unique ID of the knowledge item to be updated.
</ParamField>

### Body Parameters

<ParamField body="botIds" type="array[string]" required>
  List of bot IDs that this knowledge item is associated with.
</ParamField>

<ParamField body="status" type="string" required>
  Status of the item. Valid values: LIVE, IN_REVIEW, DRAFT.
</ParamField>

<ParamField body="question" type="string" required>
  Updated question for this knowledge item
</ParamField>

<ParamField body="answer" type="string" required>
  Updated answer for this knowledge item
</ParamField>

<ParamField body="instruction" type="string" required>
  AI agent Instructions associated with the item
</ParamField>

<ParamField body="similarQs" type="array[string]" required>
  List of similar questions to the main question.
</ParamField>

<ParamField body="keywords" type="array[string]" required>
  List of keywords to help with search and categorization.
</ParamField>

<ParamField body="categories" type="array[string]" required>
  Categories under which the item is grouped.
</ParamField>

<ParamField body="subcategories" type="array[string]" required>
  Subcategories under which the item is grouped.
</ParamField>

<ParamField body="escalation" type="boolean" required>
  Whether this item is marked for escalation.
</ParamField>

### Response

<ResponseField name="id" type="string">
  Unique ID of the updated knowledge item.
</ResponseField>

<ResponseField name="companyId" type="string">
  ID of the company that owns the knowledge item.
</ResponseField>

<ResponseField name="userId" type="string">
  ID of the user who updated the knowledge item.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Timestamp of when the item was originally created.
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Timestamp of the last update to the item.
</ResponseField>

<ResponseField name="knowledgeNumber" type="integer">
  Auto-generated knowledge number.
</ResponseField>

<ResponseField name="status" type="string">
  Status of the item: LIVE, IN_REVIEW, or DRAFT.
</ResponseField>

<ResponseField name="answer" type="string">
  The updated answer text.
</ResponseField>

<ResponseField name="question" type="string">
  The updated question text.
</ResponseField>

<ResponseField name="similarQs" type="array[string]">
  Updated list of similar questions.
</ResponseField>

<ResponseField name="keywords" type="array[string]">
  Updated keywords associated with the knowledge item.
</ResponseField>

<ResponseField name="categories" type="array[string]">
  Updated categories under which the item is grouped.
</ResponseField>

<ResponseField name="subcategories" type="array[string]">
  Updated subcategories under which the item is grouped.
</ResponseField>

<ResponseField name="escalation" type="boolean">
  Whether the item is marked for escalation.
</ResponseField>

<ResponseField name="updatedBy" type="string">
  ID of the user who last updated the item.
</ResponseField>

<ResponseField name="instruction" type="string">
  Updated instruction associated with the item.
</ResponseField>

<RequestExample>

```javascript Example Request - Node.js
const axios = require('axios');

const itemId = 'ce99afbb-9f3a-4697-81e8-4858f48895e2'; // Replace with your item ID
const url = `https://api-prod.usefini.com/v2/knowledge-items/${itemId}/public`;
const token = 'your_api_key_here'; // Replace with your actual API key

const headers = {
  'authorization': `Bearer ${token}`,
  'Content-Type': 'application/json'
};

const data = {
  botIds: ['1124e46-8dg3-5825-9a88-fb558210456t'],
  status: 'LIVE',
  question: 'Updated question',
  answer: 'Updated answer',
  instruction: 'Review this item weekly',
  similarQs: ['Updated similar question'],
  keywords: ['updated keyword'],
  categories: ['Updated Category'],
  subcategories: ['Updated Subcategory'],
  escalation: true
};

axios.put(url, data, { headers })
  .then(response => {
    console.log(response.data);
  })
  .catch(error => {
    console.error('Error:', error);
  });
```
</RequestExample>
